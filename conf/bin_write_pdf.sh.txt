#! /bin/bash
# This needs to be run by root, as wkhtmltopdf doesn't seem to work otherwise

echo 'Writing PDF...'

cd ${buildout:directory}/parts/jobs
python write_static_mlgb.py

cd ${buildout:directory}/static/media/pdf

${buildout:directory}/bin/wkhtmltopdf.sh --encoding utf8 mlgb3.html mlgb3.pdf

chown mlgb mlgb3.*
chgrp mlgb mlgb3.*
